الخطوة 4: الاتصال

وصّل الأردوينو بالكمبيوتر عبر USB

// Arduino Snake Game Controller
const int UP_BTN = 2;
const int DOWN_BTN = 3;
const int LEFT_BTN = 4;
const int RIGHT_BTN = 5;

void setup() {
  Serial.begin(9600);
  pinMode(UP_BTN, INPUT_PULLUP);
  pinMode(DOWN_BTN, INPUT_PULLUP);
  pinMode(LEFT_BTN, INPUT_PULLUP);
  pinMode(RIGHT_BTN, INPUT_PULLUP);
}

void loop() {
  if (digitalRead(UP_BTN) == LOW) { Serial.println("UP"); delay(200); }
  if (digitalRead(DOWN_BTN) == LOW) { Serial.println("DOWN"); delay(200); }
  if (digitalRead(LEFT_BTN) == LOW) { Serial.println("LEFT"); delay(200); }
  if (digitalRead(RIGHT_BTN) == LOW) { Serial.println("RIGHT"); delay(200); }
}
